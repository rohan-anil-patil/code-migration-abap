FORM SELECT_DATA .

  REFRESH IBSAD.

  SELECT * FROM BSAD
    INTO TABLE IBSAD
   WHERE BUKRS IN S_BUKRS
     AND KUNNR IN S_KUNNR
     AND BUDAT IN S_BUDAT
     AND BLART EQ 'RV'.

  IF IBSAD[] IS NOT INITIAL.
    SORT IBSAD BY KUNNR.

    REFRESH IVBRK.

    SELECT * FROM VBRK
      INTO TABLE IVBRK
      FOR ALL ENTRIES IN IBSAD
      WHERE VBELN EQ IBSAD-VBELN.

    IF IVBRK[] IS NOT INITIAL.
      LOOP AT IVBRK.
        IF IVBRK-FKART EQ 'S1'.
          DELETE TABLE IVBRK FROM IVBRK.
        ENDIF.
        IF IVBRK-FKART EQ 'ZRE'.
          DELETE TABLE IVBRK FROM IVBRK.
        ENDIF.

        IF IVBRK-FKSTO EQ 'X'.
          DELETE TABLE IVBRK FROM IVBRK.
        ENDIF.
        CLEAR IVBRK.
      ENDLOOP.
    ENDIF.
*        AND ( FKART NE 'S1' OR  FKART NE 'ZRE' )
*        AND FKSTO NE 'X'.


    LOOP AT IBSAD.
      CLEAR IVBRK.

      READ TABLE IVBRK WITH KEY VBELN = IBSAD-VBELN.
      IF SY-SUBRC = 0.
        IFINAL-COUNT = '1'.

        MOVE IBSAD-BUKRS TO IFINAL-BUKRS.
        MOVE IBSAD-KUNNR TO IFINAL-KUNNR.

        SELECT SINGLE NAME1 INTO IFINAL-NAME1
          FROM KNA1 BYPASSING BUFFER
          WHERE KUNNR = IBSAD-KUNNR.

        IF IBSAD-VBELN IS NOT INITIAL.
          CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
            EXPORTING
              INPUT  = IBSAD-VBELN
            IMPORTING
              OUTPUT = IBSAD-VBELN.

          MOVE IVBRK-BUPLA TO IFINAL-BUPLA.
*        SELECT SINGLE BUPLA  FROM VBRK
*          INTO IFINAL-BUPLA
*          WHERE VBELN = IBSAD-VBELN.

          SELECT SINGLE LIFNR  FROM VBPA
         INTO IFINAL-LIFNR
         WHERE VBELN = IBSAD-VBELN
           AND PARVW = 'VE'.

          SELECT SINGLE NAME1 FROM LFA1 INTO IFINAL-LNAME
            WHERE LIFNR = IFINAL-LIFNR.

        ENDIF.

*      IFINAL-DUE_DT = IBSAD-AUGDT  + IBSAD-ZBD1T.
        MOVE IBSAD-BELNR  TO IFINAL-BELNR.
        MOVE IBSAD-GJAHR TO IFINAL-GJAHR.
        MOVE IBSAD-AUGBL TO IFINAL-AUGBL.
        MOVE IBSAD-AUGDT TO IFINAL-AUGDT.
        MOVE IBSAD-ZUONR TO IFINAL-ZUONR.
        MOVE IBSAD-BUDAT TO IFINAL-BUDAT.
        MOVE IBSAD-BLDAT TO IFINAL-BLDAT.
        MOVE IBSAD-BLART TO IFINAL-BLART.

        IF IBSAD-SHKZG = 'H'                         .
          IBSAD-DMBTR = IBSAD-DMBTR * -1             .
        ELSE .
          IBSAD-DMBTR = IBSAD-DMBTR * 1.
        ENDIF.

        MOVE IBSAD-DMBTR TO IFINAL-DMBTR.
        MOVE IBSAD-SHKZG TO IFINAL-SHKZG.

        CLEAR I_FAEDE.
        I_FAEDE-SHKZG = IBSAD-SHKZG .
        I_FAEDE-ZFBDT = IBSAD-ZFBDT .
        I_FAEDE-ZBD1T = IBSAD-ZBD1T .
        I_FAEDE-BLDAT = IBSAD-BLDAT .
        SELECT SINGLE KOART FROM BSEG INTO I_FAEDE-KOART
          WHERE BUKRS EQ IBSAD-BUKRS
            AND BELNR EQ IBSAD-BELNR
            AND GJAHR EQ IBSAD-GJAHR
            AND AUGBL EQ IBSAD-AUGBL
            AND KUNNR EQ IBSAD-KUNNR.

        IF I_FAEDE IS NOT INITIAL.
          CLEAR E_FAEDE.
          CALL FUNCTION 'DETERMINE_DUE_DATE'
            EXPORTING
              I_FAEDE                    = I_FAEDE
*             I_GL_FAEDE                 =
            IMPORTING
              E_FAEDE                    = E_FAEDE
            EXCEPTIONS
              ACCOUNT_TYPE_NOT_SUPPORTED = 1
              OTHERS                     = 2.
          IF SY-SUBRC = 0.
            MOVE E_FAEDE-NETDT TO IFINAL-NETDT.

            CLEAR DAYS_BETWEEN.

            CALL FUNCTION 'HR_99S_INTERVAL_BETWEEN_DATES'
              EXPORTING
                BEGDA = E_FAEDE-NETDT
                ENDDA = IBSAD-AUGDT
              IMPORTING
                DAYS  = DAYS_BETWEEN.

          ENDIF.

        ENDIF.

        IF DAYS_BETWEEN GE 0.
          IN_AMT = KBETR / 365 * DAYS_BETWEEN.

          IFINAL-IN_AMT = IN_AMT * IBSAD-DMBTR / 100 ."* DAYS_BETWEEN.
        ENDIF.

*INV
        CALL FUNCTION 'HR_99S_INTERVAL_BETWEEN_DATES'
          EXPORTING
            BEGDA = IBSAD-BLDAT
            ENDDA = E_FAEDE-NETDT
          IMPORTING
            DAYS  = IFINAL-CRDYS.

        IF IFINAL-CRDYS = '1'.
          IFINAL-CRDYS = '0'.
        ENDIF.

        CALL FUNCTION 'HR_99S_INTERVAL_BETWEEN_DATES'
          EXPORTING
            BEGDA = IBSAD-BLDAT "IBSAD-AUGDT
            ENDDA = IBSAD-AUGDT "IBSAD-BLDAT
          IMPORTING
            DAYS  = IFINAL-PDDYS.

        IF IFINAL-PDDYS = '1'.
          IFINAL-PDDYS = '0'.
        ENDIF.

        IFINAL-INV =  100 * IFINAL-IN_AMT / IBSAD-DMBTR.

        MOVE IFINAL-KUNNR TO ICOLL-KUNNR.
        MOVE IFINAL-DMBTR TO ICOLL-DMBTR.
        MOVE IFINAL-IN_AMT TO ICOLL-IN_AMT.

        APPEND IFINAL.
        COLLECT  ICOLL .

        CLEAR: IBSAD,IFINAL,IN_AMT,IVBRK.
      ENDIF.
    ENDLOOP.

  ENDIF.

  LOOP AT IFINAL.
    DATA : TINV LIKE BSAD-WRBTR,
           FLG  TYPE C.
    READ TABLE ICOLL WITH KEY KUNNR = IFINAL-KUNNR.

    AT NEW KUNNR.
      CLEAR FLG.
      FLG = 'X'.
      IF ICOLL-IN_AMT IS NOT INITIAL AND ICOLL-DMBTR IS NOT INITIAL.
        TINV = 100 * ICOLL-IN_AMT / ICOLL-DMBTR.
      ENDIF.
    ENDAT.

    IF FLG = 'X'.
      IFINAL-TINV = TINV.
    ELSE.
      CLEAR IFINAL-TINV.
    ENDIF.

    MODIFY TABLE IFINAL FROM IFINAL ."TRANSPORTING TINV.
    CLEAR : IFINAL,ICOLL,TINV,FLG .
  ENDLOOP.


ENDFORM.
